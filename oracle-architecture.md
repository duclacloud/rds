## ðŸ” **Oracle Management Commands**

### **Instance and Database Information**
```sql
-- Check instance status
SELECT instance_name, status, database_status 
FROM v$instance;

-- Database information
SELECT name, dbid, created, log_mode 
FROM v$database;

-- Tablespace information
SELECT tablespace_name, status, contents, extent_management
FROM dba_tablespaces;

-- Data file locations
SELECT file_name, tablespace_name, bytes/1024/1024 as size_mb, status
FROM dba_data_files
ORDER BY tablespace_name;
```

### **Memory and Process Monitoring**
```sql
-- SGA information
SELECT component, current_size/1024/1024 as size_mb
FROM v$sga_dynamic_components
WHERE current_size > 0;

-- Active sessions
SELECT username, status, machine, program, logon_time
FROM v$session
WHERE username IS NOT NULL
ORDER BY logon_time;

-- Background processes
SELECT pname, status, description
FROM v$bgprocess
WHERE pname IS NOT NULL
ORDER BY pname;
```

### **Storage and Performance**
```sql
-- Table sizes
SELECT owner, table_name, 
       ROUND(bytes/1024/1024,2) as size_mb,
       num_rows
FROM dba_segments s, dba_tables t
WHERE s.owner = t.owner 
  AND s.segment_name = t.table_name
  AND s.owner NOT IN ('SYS','SYSTEM')
ORDER BY bytes DESC;

-- Index usage
SELECT owner, index_name, table_name, status, 
       ROUND(bytes/1024/1024,2) as size_mb
FROM dba_indexes i, dba_segments s
WHERE i.owner = s.owner
  AND i.index_name = s.segment_name
  AND i.owner NOT IN ('SYS','SYSTEM')
ORDER BY bytes DESC;

-- Wait events
SELECT event, total_waits, time_waited, avg_wait
FROM v$system_event
WHERE wait_class != 'Idle'
ORDER BY time_waited DESC;
```

### **Backup and Recovery**
```sql
-- Archive log information
SELECT sequence#, first_time, next_time, archived, status
FROM v$log
ORDER BY sequence#;

-- RMAN backup information
SELECT session_key, input_type, status, 
       start_time, end_time, 
       input_bytes/1024/1024 as input_mb,
       output_bytes/1024/1024 as output_mb
FROM v$rman_backup_job_details
WHERE start_time > SYSDATE - 7
ORDER BY start_time DESC;

-- Recovery area usage
SELECT space_limit/1024/1024/1024 as limit_gb,
       space_used/1024/1024/1024 as used_gb,
       space_reclaimable/1024/1024/1024 as reclaimable_gb,
       number_of_files
FROM v$recovery_file_dest;
```
